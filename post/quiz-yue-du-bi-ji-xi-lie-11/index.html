2020-06-29

原文链接

trait Trait {
    fn f(&self);
}

impl<'a> dyn Trait + 'a {
    fn f(&self) {
        print!("1");
    }
}

impl Trait for bool {
    fn f(&self) {
        print!("2");
    }
}

fn main() {
    Trait::f(&true);
    Trait::f(&true as &dyn Trait);
    <_ as Trait>::f(&true);
    <_ as Trait>::f(&true as &dyn Trait);
    <bool as Trait>::f(&true);
}

解析

NO.1

调用Trait这个trait 为bool类型实现的f方法

Trait::f(&true);

示例如下 
trait Trait1{
    fn do_something(&self);
}

trait Trait2{
    fn do_something(&self);
}

struct S;
impl Trait1 for S{
    fn do_something(&self){
        println!("Trait1");
    }
}

impl Trait2 for S{
    fn do_something(&self){
        println!("Trait2");
    }
}


fn main() {
    //Trait1
    Trait1::do_something(&S);
    //Trait2
    Trait2::do_something(&S);
}

NO.2

trait 对象内置的方法会被具体类型实现的方法遮蔽

Trait::f(&true as &dyn Trait);

NO.3

消除歧义的完全限定语法用来调用trait内部的关联函数或者调用指定trait实现的方法。

<_ as Trait>::f(&true);
//内部方法被遮蔽
<_ as Trait>::f(&true as &dyn Trait);

示例如下：

trait Trait1{
    fn do_something(){
        println!("trait 1 do something")
    }
}


struct S;
impl Trait1 for S{}
impl S{
    fn do_something(){
        println!("S do something")
    }
}
fn main() {
    S::do_something();
    <S as Trait1>::do_something();
}
