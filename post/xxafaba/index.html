trait Or {
    fn f(self);
}

struct T;

impl Or for &T {
    fn f(self) {
        print!("1");
    }
}

impl Or for &&&&T {
    fn f(self) {
        print!("2");
    }
}

fn main() {
    let t = T;
    let wt = &T;
    let wwt = &&T;
    let wwwt = &&&T;
    let wwwwt = &&&&T;
    let wwwwwt = &&&&&T;
    t.f();
    wt.f();
    wwt.f();
    wwwt.f();
    wwwwt.f();
    wwwwwt.f();
}

解析

NO.1

尝试寻找T对应的f方法，没有找到，然后给T加上&，找到对应的方法

let t = T;
t.f();

NO.2

直接找到对应的类型&T

let wt = &T;
wt.f();

NO.3

找到对应的类型&&T,没找到,加上&,还是没有，然后解引用，找到&T

let wwt = &&T;
wwt.f();

NO.4

找到对应的类型&&&T,没找到,加上&,找到 &&&&T

let wwwt = &&&T;
wwwt.f();

NO.5

找到对应的类型&&&T,没找到,加上&,没找到,解引用找到

let wwwwt = &&&&T;
wwwwt.f();

NO.6

找到对应的类型&&&&&T,没找到,加上&, 没找到,解引用找到

let wwwwwt = &&&&&T;
wwwwwt.f();
