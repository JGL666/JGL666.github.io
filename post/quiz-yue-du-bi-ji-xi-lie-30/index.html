Rust Quiz 解读汇总

use std::rc::Rc;

struct A;

fn p<X>(x: X) {
    match std::mem::size_of::<X>() {
        0 => print!("0"),
        _ => print!("1"),
    }
}

fn main() {
    let a = &A;
    p(a);
    p(a.clone());
    
    let b = &();
    p(b);
    p(b.clone());
    
    let c = Rc::new(());
    p(Rc::clone(&c));
    p(c.clone());
}

解析

NO.1

P(a) 是std::mem::size_of::<&A>(),输出1。

    let a = &A;
    p(a);
    p(a.clone());

clone的方法实现

fn clone(&self) -> Self {
    *self
}

没有为A实现clone方法，会自动给A加&。Clone默认给&T实现clone方法

impl<T: ?Sized> Clone for &T {
    #[inline]
    fn clone(&self) -> Self {
        *self
    }
}

如果实现A的clone方法，输出0

use std::clone::Clone;

impl Clone for A{
    fn clone(&self)->Self{
        A
    }
}

NO.2

b的类似是&(),p(b) 执行std::mem::size_of::<&()>()输出1，()有实现clone，所以b.clone 返回0

let b = &();
p(b);
p(b.clone());

NO.3

Rc::clone返回的是Rc,是指针，占一个指针大小

let c = Rc::new(());
p(Rc::clone(&c));
p(c.clone());
