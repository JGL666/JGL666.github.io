Rust Quiz 解读汇总

16

fn main() {
    let mut x = 4;
    --x;
    print!("{}{}", --x, --x);
}

相当于

print!("{}{}", -(-x), -(-x));

17

    let mut a = 5;
    let mut b = 3;
    print!("{}", a-- - --b);
}

上面相当于

    let mut a = 5;
    let mut b = 3;
    print!("{}", a-(-(-(-(-b)))));
