Rust Quiz 解读汇总

fn return1() {
    if (return { print!("1") }) {
    }
}

fn return2() {
    if return { print!("2") } {
    }
}

fn break1() {
    loop {
        if (break { print!("1") }) {
        }
    }
}

fn break2() {
    loop {
        if break { print!("2") } {
        }
    }
}

fn main() {
    return1();
    return2();
    break1();
    break2();
}

解析

参照表达式优先级

NO.1

return 总是优化处理后面的表达式

fn return1() {
    if (return { print!("1") }) {
    }
}

fn return2() {
    if return { print!("2") } {
    }
}

NO.2

() 优先级大于break，先计算括号表达式，输出1

fn break1() {
    loop {
        if (break { print!("1") }) {
        }
    }
}

NO.3

fn break2() {
    loop {
        if break { print!("2") } {
        }
    }
}

和return 不一样，break不优先处理后面的表达式，上面的函数格式化如下。break在 {println!("2")} 之前就退出了。

fn break2() {
    loop {
        if break
        { print!("2") }
        {}
    }
}
