Rust Quiz 解读汇总

struct S;

impl Drop for S {
    fn drop(&mut self) {
        print!("1");
    }
}

fn main() {
    let s = S;
    let _ = s;
    print!("2");
}

解析

所有权发生转移

let s = S;

所有权没有接收者，s会一直存在直到作用域结束

 let _ = s;

拓展

输出112，因为所有权没有接收者，直接drop了

struct S;

impl Drop for S {
    fn drop(&mut self) {
        print!("1");
    }
}

fn get()->S{
    S
}

fn main() {
    let _ = get();
    let _ = S;
    print!("2");
}
